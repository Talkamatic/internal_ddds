what time is it :: ask(?X.current_time(X))
what time is it please :: ask(?X.current_time(X))

set the time :: request(set_time)
set the time please :: request(set_time)

set the time to eleven ten :: request(set_time) ; answer(hour_to_set(11)) ; answer(minute_to_set(10))
set the time to eleven ten please :: request(set_time) ; answer(hour_to_set(11)) ; answer(minute_to_set(10))

ten :: answer(10)
eleven :: answer(11)
thirty five :: answer(35)
fifty :: answer(50)
